id				BIGINT		NO		Primary key
occasion_id		BIGINT		NO		Foreign key referencing the event/occasion this template belongs to
name			STRING(150)	NO		Internal name of the template
channel_id		BIGINT		NO		ID of the messaging channel (WhatsApp, RCS, etc.)
language_code	STRING(10)	NO		Template language (e.g., en_US)
category		STRING(50)	YES		Category of the template (Marketing, Utility, etc.)
-template_type	STRING(50)	YES		High-level template type (INVITATION, REMINDER, NOTIFICATION)
-components		JSON		YES		JSON array defining template components (header, body, footer, buttons)
-placeholders	JSON		YES		JSON array of variables for dynamic substitution (e.g., guest_name, event_name)
sender_id		STRING(50)	YES		WhatsApp Business number ID used to send this template
media_url		STRING(500)	YES		Permanent media link (S3/CDN) for uploading to WhatsApp dynamically
-media_id		STRING(100)	YES		Cached WhatsApp media ID for reuse; expires in 30 days
template_code	STRING		NO		actual template code	
-media_uploaded_at	DATE		YES		Timestamp of last media upload to WhatsApp
status			TINYINT		NO		Tracks approval workflow and active/inactive state
created_at		DATE		NO		Auto-generated timestamp
updated_at		DATE		NO		Auto-generated timestamp
deleted_at		DATE		YES		Soft delete timestamp
